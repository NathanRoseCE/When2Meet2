function toggleColor(btn, highColor, lowColor) {
	var property = btn;
    if (property.toggled !== 'toggled') {
        property.style.backgroundColor=lowColor;
        property.toggled = 'toggled';
    }
    else {
        property.style.backgroundColor = highColor;
        property.toggled = '';
    }
}
